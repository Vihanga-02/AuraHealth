version: '3.8'

services:
  # --- Payment Database ---
  payment-db:
    image: postgres:15
    container_name: payment-db
    environment:
      POSTGRES_USER: user123
      POSTGRES_PASSWORD: password123
      POSTGRES_DB: payment_db
    ports:
      - "5434:5432" 
      
    volumes:
      - payment_db_data:/var/lib/postgresql/data
    networks:
      - mediconnect-network

  # --- Notification Database ---
  notification-db:
    image: postgres:15
    container_name: notification-db
    environment:
      POSTGRES_USER: user123
      POSTGRES_PASSWORD: password123
      POSTGRES_DB: notification_db
    ports:
      - "5433:5432"  
    volumes:
      - notification_db_data:/var/lib/postgresql/data
    networks:
      - mediconnect-network

  # --- Payment Service ---
  payment-service:
    build: ./payment-service
    container_name: payment-service
    ports:
      - "5005:5005"
    env_file:
      - ./payment-service/.env  
    depends_on:
      - payment-db
    networks:
      - mediconnect-network

  # --- Notification Service ---
  notification-service:
    build: ./notification-service
    container_name: notification-service
    ports:
      - "5006:5006"
    env_file:
      - ./notification-service/.env
    depends_on:
      - notification-db
    networks:
      - mediconnect-network

networks:
  mediconnect-network:
    driver: bridge

volumes:
  payment_db_data:
  notification_db_data:
